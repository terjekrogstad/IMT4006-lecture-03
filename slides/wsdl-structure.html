WSDL Structure

    
    //Description document:
    <description targetNamespace="…">

    //Message formats (schema types):
      <types> … </types>

    //Abstract interface of the service:
      <interface>
        <operation> … </operation>+
      </interface>+

    //Binding of these messages to a protocol:
      <binding> … </binding>+

    //Location of the service:
      <service>
        <endpoint> … </endpoint>
      </service>

    </description>

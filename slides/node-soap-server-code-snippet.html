Node-Soap Server Code Snippet

    Example using Node-soap.
    
  var myService = {
      MyService: {
          MyPort: {
              GetCityWeatherByZIP: function(args) {
                  var responseData = '';

                  //Do some fancy backend stuff...

                  return responseData;
              }
          }
      }
  }

  var xml = require('fs').readFileSync('myservice.wsdl', 'utf8'),
      server = http.createServer(function(request,response) {
          response.end( "404: Not Found: " + request.url)
      });

  server.listen(8000);
  soap.listen(server, '/wsdl', myService, xml);

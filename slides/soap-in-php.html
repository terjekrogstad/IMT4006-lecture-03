SOAP in PHP

    Client example with NuSOAP
    
    require('nusoap.php');
    $client = new soapClient('http://myservice.com/xml/CDdbService.php');

    // Parameters used in the SOAP request
    $perfName = 'Pink Floyd';
    $albName  = 'The Wall';

    $paramsAlbum = array('performerName' => $perfName,
                         'albumName' => $albName);
       
    // Marshalling parameters, operation request, unmarshalling return value
    $result = $client->call( 'getProductionYear', $paramsAlbum );
    if (!$err = $client->getError()) {
      // Year of production returned
      echo 'Production year: ' . $result;
    } else {
      echo 'Request failed: ' . $err;
    }
